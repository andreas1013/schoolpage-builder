// ============================================================ // SchoolPage Builder — Cloudflare Worker // Routes: // POST / → Anthropic API proxy // POST /scrape → Scrape a Queensland school website page // POST /scrape-page → Scrape a single page for full content // ============================================================ const ANTHROPIC_API_KEY = "YOUR_KEY_HERE"; export default { async fetch(request) { if (request.method === "OPTIONS") { return new Response(null, { status: 204, headers: cors() }); } const url = new URL(request.url); // ── ROUTE: POST / → Anthropic proxy ── if (url.pathname === "/" && request.method === "POST") { return handleAnthropic(request); } // ── ROUTE: POST /scrape → Fetch nav + page list from school homepage ── if (url.pathname === "/scrape" && request.method === "POST") { return handleScrape(request); } // ── ROUTE: POST /scrape-page → Fetch full content of one page ── if (url.pathname === "/scrape-page" && request.method === "POST") { return handleScrapePage(request); } return new Response(JSON.stringify({ error: "Not found" }), { status: 404, headers: cors() }); } }; // ── ANTHROPIC PROXY ── async function handleAnthropic(request) { let body; try { body = await request.json(); } catch { return json({ error: "Invalid JSON" }, 400); } const res = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": ANTHROPIC_API_KEY, "anthropic-version": "2023-06-01", }, body: JSON.stringify(body), }); const data = await res.json(); return json(data, res.status); } // ── SCRAPE HOMEPAGE: get nav structure ── async function handleScrape(request) { let body; try { body = await request.json(); } catch { return json({ error: "Invalid JSON" }, 400); } const siteUrl = body.url; if (!siteUrl) return json({ error: "No URL provided" }, 400); // Validate it's a Queensland school or reasonable education URL let parsedUrl; try { parsedUrl = new URL(siteUrl); } catch { return json({ error: "Invalid URL" }, 400); } try { const res = await fetch(siteUrl, { headers: { "User-Agent": "Mozilla/5.0 (compatible; SchoolPageBuilder/1.0)", "Accept": "text/html" }, redirect: "follow" }); if (!res.ok) return json({ error: "Could not fetch site (status " + res.status + ")" }, 400); const html = await res.text(); const nav = extractNav(html, parsedUrl.origin); return json({ success: true, nav: nav, origin: parsedUrl.origin }); } catch (e) { return json({ error: "Failed to fetch: " + e.message }, 500); } } // ── SCRAPE SINGLE PAGE: get full content ── async function handleScrapePage(request) { let body; try { body = await request.json(); } catch { return json({ error: "Invalid JSON" }, 400); } const pageUrl = body.url; if (!pageUrl) return json({ error: "No URL provided" }, 400); try { const res = await fetch(pageUrl, { headers: { "User-Agent": "Mozilla/5.0 (compatible; SchoolPageBuilder/1.0)", "Accept": "text/html" }, redirect: "follow" }); if (!res.ok) return json({ error: "Could not fetch page (status " + res.status + ")" }, 400); const html = await res.text(); const data = extractPageContent(html, pageUrl); return json({ success: true, ...data }); } catch (e) { return json({ error: "Failed to fetch page: " + e.message }, 500); } } // ── HTML PARSER HELPERS ── // Extract navigation structure from the HTML function extractNav(html, origin) { const pages = []; const seen = new Set(); // Queensland DET school sites (Squiz Matrix) patterns // Top nav const topNavPatterns = [ /]*(?:id|class)="[^"]*(?:top|global|main|primary|site)[^"]*"[^>]*>([\s\S]*?)<\/nav>/gi, /]*(?:id|class)="[^"]*(?:top|global|main|primary|nav|menu)[^"]*"[^>]*>([\s\S]*?)<\/ul>/gi, /]*(?:id|class)="[^"]*(?:top-nav|main-nav|primary-nav|navigation)[^"]*"[^>]*>([\s\S]*?)<\/div>/gi, ]; // Left sidebar nav (common on QLD school sites) const sideNavPatterns = [ /<(?:nav|ul|div)[^>]*(?:id|class)="[^"]*(?:side|left|local|section|sub)[^"]*nav[^"]*"[^>]*>([\s\S]*?)<\/(?:nav|ul|div)>/gi, /<(?:nav|ul|div)[^>]*(?:id|class)="[^"]*nav[^"]*(?:side|left|local|section|sub)[^"]*"[^>]*>([\s\S]*?)<\/(?:nav|ul|div)>/gi, ]; // Also grab all internal links as fallback const allPatterns = [...topNavPatterns, ...sideNavPatterns]; for (const pattern of allPatterns) { let match; while ((match = pattern.exec(html)) !== null) { const section = match[1] || match[0]; extractLinksFromHtml(section, origin, seen, pages); } pattern.lastIndex = 0; } // If nothing found, grab ALL internal links from the page if (pages.length === 0) { extractLinksFromHtml(html, origin, seen, pages); } // Sort and deduplicate return pages .filter(p => p.url && p.title) .slice(0, 60); // cap at 60 pages } function extractLinksFromHtml(html, origin, seen, pages) { const linkPattern = /]+href="([^"#]+)"[^>]*>([\s\S]*?)<\/a>/gi; let m; while ((m = linkPattern.exec(html)) !== null) { let href = m[1].trim(); let rawTitle = m[2].replace(/<[^>]+>/g, '').trim(); if (!rawTitle || rawTitle.length < 2) continue; if (rawTitle.length > 80) continue; // skip long link texts // Resolve relative URLs try { if (href.startsWith('/')) href = origin + href; else if (!href.startsWith('http')) href = origin + '/' + href; const u = new URL(href); // Only include internal links if (!href.startsWith(origin) && u.hostname !== new URL(origin).hostname) continue; // Skip non-page links if (/\.(pdf|doc|docx|xls|xlsx|jpg|jpeg|png|gif|mp4|zip)$/i.test(href)) continue; if (seen.has(href)) continue; seen.add(href); pages.push({ url: href, title: rawTitle }); } catch (e) { continue; } } } // Extract content from a single page function extractPageContent(html, pageUrl) { // Page title let title = ''; const titleMatch = html.match(/]*>([^<]+)<\/title>/i); if (titleMatch) title = titleMatch[1].replace(/\s*[-|]\s*.+$/, '').trim(); // h1 as a better title const h1Match = html.match(/]*>([\s\S]*?)<\/h1>/i); if (h1Match) title = h1Match[1].replace(/<[^>]+>/g, '').trim() || title; // Main content area — try common patterns let contentHtml = ''; const contentPatterns = [ /]*>([\s\S]*?)<\/main>/i, /]*(?:id|class)="[^"]*(?:main-content|page-content|content-area|content-body|entry-content)[^"]*"[^>]*>([\s\S]*?)<\/div>/i, /]*>([\s\S]*?)<\/article>/i, /]*(?:id|class)="[^"]*content[^"]*"[^>]*>([\s\S]*?)<\/div>/i, ]; for (const p of contentPatterns) { const m = html.match(p); if (m && m[1] && m[1].length > 200) { contentHtml = m[1]; break; } } if (!contentHtml) contentHtml = html; // fallback to full page // Extract clean text from content const text = htmlToText(contentHtml); // Extract links from content const links = []; const seen = new Set(); const lp = /]+href="([^"#]+)"[^>]*>([\s\S]*?)<\/a>/gi; let m; while ((m = lp.exec(contentHtml)) !== null) { const href = m[1].trim(); const label = m[2].replace(/<[^>]+>/g, '').trim(); if (!label || label.length < 2 || seen.has(href)) continue; if (/^(mailto:|tel:|javascript:)/i.test(href)) continue; seen.add(href); links.push({ url: href, label: label }); } return { title, text, links: links.slice(0, 30) }; } function htmlToText(html) { // Remove scripts, styles, nav, header, footer html = html.replace(/<(script|style|nav|header|footer)[^>]*>[\s\S]*?<\/\1>/gi, ''); // Replace block elements with newlines html = html.replace(/<\/(p|div|li|h[1-6]|br|tr)>/gi, '\n'); html = html.replace(//gi, '\n'); // Remove all remaining tags html = html.replace(/<[^>]+>/g, ''); // Decode common entities html = html.replace(/&/g, '&').replace(/</g, '<').replace(/>/g, '>') .replace(/ /g, ' ').replace(/"/g, '"').replace(/'/g, "'"); // Clean up whitespace html = html.split('\n').map(l => l.trim()).filter(l => l.length > 0).join('\n'); html = html.replace(/\n{3,}/g, '\n\n'); return html.trim().substring(0, 8000); // cap at 8000 chars } // ── HELPERS ── function json(data, status) { return new Response(JSON.stringify(data), { status: status || 200, headers: cors() }); } function cors() { return { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type", }; }
